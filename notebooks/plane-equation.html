P0 = np.array([1, 1, 1])
n = np.array([1, 0, 1])

# d = -(ax_0 + by_0 + cz_0)
d = -np.dot(n, P0)

num_points = 3000
x = np.random.uniform(low=-3, high=3, size=num_points)
y = np.random.uniform(low=-3, high=3, size=num_points)

# ax + by + cz + d = 0
# cz = -(ax + by + d)
# z = -(ax + by + d) / c
a, b, c = n
z = -(a * x + b * y + d) / c

P = np.hstack([x[None, :].T, y[None, :].T, z[None, :].T])
p = P - P0

assert np.allclose(np.dot(p, n), 0)

    P0[2],
